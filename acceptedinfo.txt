#include <bits/stdc++.h>
using namespace std;
#define int long long

#include "mydebug.h"
#else
#define debug(...) 1
#endif

const int N = 105;
int dp[N][N], arr[N], w, k;

int f(int ind, int num) {
    if (ind < 0 || num == -1) return 0;
    if (dp[ind][num] != -1) return dp[ind][num];
    int cnt = 0;
    for (int i = ind; i >= 0; i--) {
        if(arr[ind] - w > arr[i]) break;
        cnt++;
        // debug(cnt, i);
    }
    int x = max(cnt + f(ind-cnt, num - 1), f(ind - 1, num));
    return dp[ind][num] = x;
}

void solve() {
    memset(dp, -1, sizeof dp);
    int n;
    cin >> n >> w >> k;
    for (int i = 0; i < n; i++) {
        int x;
        cin >> x >> arr[i];
    }
    sort(arr, arr + n);
    cout << f(n - 1, k - 1) << "\n";
}

int32_t main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int n;
    cin >> n;
    for (int i = 1; i <= n; i++) {
        cout << "Case " << i << ": ";
        solve();
    }
}
